# import math

def calculate_crack_time(password_length, char_set_size, attack_speed):
    """
    Estimate the time required to crack a password.
    
    Parameters:
    password_length (int): Length of the password.
    char_set_size (int): Size of the character set (e.g., 26 for lowercase letters).
    attack_speed (float): Number of guesses per second.
    
    Returns:
    float: Time in seconds to crack the password.
    """
    # Total number of possible passwords
    total_passwords = char_set_size ** password_length
    
    # Time to crack in seconds
    time_to_crack = total_passwords / attack_speed
    
    return time_to_crack

def display_time(seconds):
    """
    Display the time in a human-readable format.
    
    Parameters:
    seconds (float): Time in seconds.
    """
    if seconds < 60:
        print(f"Time to crack: {seconds:.2f} seconds")
    elif seconds < 3600:
        print(f"Time to crack: {seconds / 60:.2f} minutes")
    elif seconds < 86400:
        print(f"Time to crack: {seconds / 3600:.2f} hours")
    else:
        print(f"Time to crack: {seconds / 86400:.2f} days")

def main():
    print("Welcome to the Non-Hashed Password Cracker Simulation!")
    
    # User input
    password_length = int(input("Enter password length: "))
    char_set_size = int(input("Enter size of character set (e.g., 26 for lowercase letters): "))
    attack_speed = float(input("Enter attack speed (guesses per second): "))
    
    # Calculate and display cracking time
    time_to_crack = calculate_crack_time(password_length, char_set_size, attack_speed)
    display_time(time_to_crack)

if __name__ == "__main__":
    main()
